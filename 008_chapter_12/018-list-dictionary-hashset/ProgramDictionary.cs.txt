// Пример использования Dictionary<TKey, TValue>
Console.WriteLine("\nDictionary<TKey, TValue> пример:");
// Dictionary<string, int> ages = new Dictionary<string, int>()
// Dictionary<string, int> ages = new()
//     {
//         { "Alice", 30 }, // инициализация
//         { "Bob", 25 }
//     };
Dictionary<string, int> ages = new(); // new() - вызов конструктора для размещения в памяти
ages.Add("Alice", 30);
ages.Add("Bob", 25);
// Добавление новой пары ключ-значение
ages["Charlie"] = 28;
ages.Remove("Bob"); // удаление значений происходит по ключу

// foreach (KeyValuePair<string, int> pair in ages)
foreach (var pair in ages)
{
    Console.WriteLine($"Имя: {pair.Key}, Возраст: {pair.Value}"); // обращение к основным свойствам
}

// foreach (var item in ages.Keys) // коллекций ключей
foreach (var item in ages.Values) // коллекция значений
{
    Console.WriteLine("Возраст: " + item);
}
